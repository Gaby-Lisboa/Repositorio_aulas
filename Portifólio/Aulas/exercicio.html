Exercícios de JavaScript

Exercício 1: Diferença entre dois números

//comentário
var num1 = Number(document.getElementById("num1_1").value);
var num2 = Number(document.getElementById("num2_1").value);
var diferenca = num1 - num2;
document.getElementById("result1").innerText = `A diferença entre ${num1} e ${num2} é de ${diferenca}`;

Exercício 2: Verificar se um número é par

var num = Number(document.getElementById("num2").value);
if (num % 2 === 0) {
document.getElementById("result2").innerText = `O número ${num} é par`;
} else {
document.getElementById("result2").innerText = `O número ${num} é ímpar`;
}

Exercício 3: Múltiplo do Segundo Número

var nota1 = Number(document.getElementById("num1_3").value);
var nota2 = Number(document.getElementById("num2_3").value);
if (nota1 % nota2 === 0) {
document.getElementById("result3").innerText = `${nota1} é múltiplo de ${nota2}`;
} else {
document.getElementById("result3").innerText = `${nota1} não é múltiplo de ${nota2}`;
}

Exercício 4: Converter Celsius para Fahrenheit

var celsius = Number(document.getElementById("celsius").value);
var fahrenheit = (celsius * 9/5) + 32;
document.getElementById("result4").innerText = `${celsius}°C é equivalente a ${fahrenheit}°F`;
